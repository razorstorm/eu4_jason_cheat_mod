country_decisions = {

	set_gold_dev = {
		major = yes
		potential = {
			ai = no
		}
		provinces_to_highlight = {
			OR = {
				trade_goods = gold
				trade_goods = silver
			}
		}
		effect = {
			country_event = { 
				id = gold_upgrade_events.1
			}
		}
	}

	upgrade_gold_dev = {
		major = yes
		potential = {
			ai = no
		}
		provinces_to_highlight = {
			OR = {
				trade_goods = gold
				trade_goods = silver
			}
		}
		effect = {
			country_event = { id = gold_upgrade_events.2 } 
		}
	}

	upgrade_silk_dev = {
		major = yes
		potential = {
			ai = no
		}
		provinces_to_highlight = {
			OR = {
				trade_goods = silk
				trade_goods = dyes
			}
		}
		effect = {
			every_owned_province = {
				limit = {
					OR = {
						trade_goods = silk
						trade_goods = dyes
					}
				}
				add_base_tax = 2
				add_base_production = 3
				add_base_manpower = 1
			}
		}
	}

	set_silk_dev = {
		major = yes
		potential = {
			ai = no
		}
		provinces_to_highlight = {
			OR = {
				trade_goods = silk
				trade_goods = dyes
			}
		}
		effect = {
			every_owned_province = {
			limit = {
				OR = {
					trade_goods = silk
					trade_goods = dyes
				}
				NOT = {
					# Ignore provinces that are already highly devved 
					base_production = 25
				}
			}
			if = {
				set_base_tax = 20
				set_base_production = 25
				set_base_manpower = 15
			}
		}
	}


	discount_gold_dev = {
		major = yes
		potential = {
			ai = no
		}
		provinces_to_highlight = {
			OR = {
				trade_goods = gold
				trade_goods = silver
			}
		}
		effect = {
			country_event = { id = gold_upgrade_events.3 } 
		}
	}

}
