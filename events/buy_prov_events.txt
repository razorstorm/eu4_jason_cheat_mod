# ROOT is marked as 'selected_neighbor_country'
country_event = {
	id = buy_prov_events.1
	title = "buy_prov_events.EVTNAME1"
	desc = "buy_prov_events.EVTDESC1"
	picture = CITY_VIEW_eventPicture

	hidden = yes
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			save_global_event_target_as = selected_neighbor_country
			every_owned_province = {
				limit = {
					OR = {
						AND = {
							has_port = yes
							sea_zone = {
								any_neighbor_province = {
									has_port = yes
									OR = {
										AND = {
											owned_by = FROM
											is_core = FROM
										}
										AND = {
											owner = { is_colonial_nation_of = FROM }
											is_core = owner
										}
									}
								}
							}
						}
						any_neighbor_province = {
							OR = {
								AND = {
									owned_by = FROM
									is_core = FROM
								}
								AND = {
									owner = { is_colonial_nation_of = FROM }
									is_core = owner
								}
							}
						}
					}
				}
				set_province_flag = counted_as_possible_province

				# Calculate additional price for province, which has to be paid, if province has much trade influence
				# Export province's base trade power
				export_to_variable = {
					which = province_trade_power_variable
					value = modifier:province_trade_power_value
				}
				# Export province's trade power modifier
				export_to_variable = {
					which = province_trade_power_modifier_variable
					value = modifier:province_trade_power_modifier
				}
				# Export province's owner's trade power modifier
				export_to_variable = {
					which = owner_trade_power_modifier_variable
					value = modifier:global_prov_trade_power_modifier
					who = ROOT
				}
				# Add 1.0 to province's trade power modifier, because it is 100% by default
				change_variable = { which = province_trade_power_modifier_variable value = 1.0 }
				# Add both trade power modifier variables
				change_variable = { which = province_trade_power_modifier_variable which = owner_trade_power_modifier_variable }
				# Multiply province's base trade power with total trade power modifier to get current total province trade power
				multiply_variable = { which = province_trade_power_variable which = province_trade_power_modifier_variable }

				# Round variable
				change_variable = { which = province_trade_power_variable value = 0.5 }
				divide_variable = { which = province_trade_power_variable value = 1000 }
				multiply_variable = { which = province_trade_power_variable value = 1000 }

				# Multiply current total province trade power with VALUE to get additional price
				multiply_variable = { which = province_trade_power_variable value = 10 }
			}
			FROM = { country_event = { id = buy_prov_events.3 } }	# AI accepted
		}
	}

	option = {
		name = "buy_prov_events1.EVTOPT1"
		ai_chance = { factor = 100 }
	}
}
